 14class ScaleRange:
 15    '''
 16    If `plotting.normalize=True`, the data will be normalized inside the specified range.
 17    The vertical scale of the plots is still determined by the full data range, unless
 18    `plotting.zoom_range=True`, so that the data is scaled inside the range.
 19    If only `plotting.zoom_range=True`, with `plotting.normalize=False`, only the `index` dataset
 20    will be scaled to fit the range, and the rest of the datasets will be scaled accordingly. 
 21    '''
 22    def __init__(self,
 23                 index:int=0,
 24                 xmin:float=None,
 25                 xmax:float=None,
 26                 ymin:list=None,
 27                 ymax:list=None,
 28                 ):
 29        self.index = index
 30        '''Index of the dataframe to use as reference.'''
 31        self.xmin = xmin
 32        self.xmax = xmax
 33        self.ymin = ymin
 34        self.ymax = ymax
 35        '''If `plotting.normalize=True`, normalize the plots according to the y-values provided.'''
 36    def x(self, xmin:float=None, xmax:float=None):
 37        self.xmin = xmin
 38        self.xmax = xmax
 39        self.ymin = None
 40        self.ymax = None
 41        return self
 42    def y(self, ymin:list=None, ymax:list=None):
 43        self.xmin = None
 44        self.xmax = None
 45        self.ymin = ymin
 46        self.ymax = ymax
 47        return self

170            self,
171            units,
172            units_in=None,
173            default_unit='cm-1',
174            ):
175        ''''Method to change units. ALWAYS use this method to do that.'''
176
177        mev = 'meV'
178        cm = 'cm-1'
179        unit_format={
180                mev: ['mev', 'meV', 'MEV'],
181                cm: ['cm', 'CM', 'cm-1', 'cm^-1'],
182            }
183
184        if self.units is not None:
185            units_in = deepcopy(self.units)
186            self.units = units
187        elif units is not None:
188            units_in = units_in
189            self.units = deepcopy(units)
190        elif units is None and units_in is None:
191            units_in = None
192            self.units = default_unit
193        elif units is None and units_in is not None:
194            units_in = None
195            self.units = deepcopy(units_in)
196
197        if isinstance(units_in, list):
198            for i, unit_in in enumerate(units_in):
199                for key, value in unit_format.items():
200                    if unit_in in value:
201                        units_in[i] = key
202                        break
203            if len(units_in) == 1:
204                units_in = units_in * len(self.filename)
205            elif len(units_in) != len(self.filename):
206                raise ValueError("units_in must be a list of the same length as filenames.")
207
208        if isinstance(units_in, str):
209            for key, value in unit_format.items():
210                if units_in in value:
211                    units_in = key
212                    break
213            units_in = [units_in] * len(self.filename)
214        
215        if isinstance(self.units, list):
216            for i, unit in enumerate(self.units):
217                for key, value in unit_format.items():
218                    if unit in value:
219                        self.units[i] = key
220                        break
221            if len(self.units) == 1:
222                self.units = self.units * len(self.filename)
223            elif len(self.units) != len(self.filename):
224                raise ValueError("units_in must be a list of the same length as filenames.")
225
226        if isinstance(self.units, str):
227            for key, value in unit_format.items():
228                if self.units in value:
229                    self.units = key
230                    break
231            self.units = [self.units] * len(self.filename)
232        
233        if units_in is None:
234            return self
235        # Otherwise, convert the dataframes
236        if len(self.units) != len(units_in):
237            raise ValueError("Units len mismatching.")
238        for i, unit in enumerate(self.units):
239            if unit == units_in[i]:
240                continue
241            if unit == mev and units_in[i] == cm:
242                self.dataframe[i][self.dataframe[i].columns[0]] = self.dataframe[i][self.dataframe[i].columns[0]] * cm_to_meV
243            elif unit == cm and units_in[i] == mev:
244                self.dataframe[i][self.dataframe[i].columns[0]] = self.dataframe[i][self.dataframe[i].columns[0]] * meV_to_cm
245            else:
246                raise ValueError(f"Unit conversion error between '{unit}' and '{units_in[i]}'")
247
248        # Rename dataframe columns
249        E_units = None
250        for i, df in enumerate(self.dataframe):
251            if self.units[i] == mev:
252                E_units = 'meV'
253            elif self.units[i] == cm:
254                E_units = 'cm-1'
255            else:
256                E_units = self.units[i]
257
258            if self.type == 'INS':
259                if self.dataframe[i].shape[1] == 3:
260                    self.dataframe[i].columns = [f'Energy transfer / {E_units}', 'S(Q,E)', 'Error']
261                else:
262                    self.dataframe[i].columns = [f'Energy transfer / {E_units}', 'S(Q,E)']
263            elif self.type == 'ATR':
264                self.dataframe[i].columns = [f'Wavenumber / {E_units}', 'Absorbance']
265            elif self.type == 'RAMAN':
266                self.dataframe[i].columns = [f'Raman shift / {E_units}', 'Counts']
267
268        return self

311    def set_grams_error(self):
312        '''Set the error in grams, based on the number of decimal places.'''
313        if self.grams is None:
314            return
315        decimal_accuracy = len(str(self.grams).split('.')[1])
316        # Calculate the error in grams
317        self.grams_error = 10**(-decimal_accuracy)
318
319    def set_mass(self):
320        '''Set the molar mass of the material.\n
321        If `self.grams` is not `None`, the number of moles will be calculated and overwritten.'''
322        material_grams_per_mol = 0.0
323        for key in self.atoms:
324            material_grams_per_mol += self.atoms[key] * mass[key]
325        self.molar_mass = material_grams_per_mol
326        if self.grams is not None:
327            self.set_grams_error()
328            self.mols = self.grams / material_grams_per_mol
329            self.mols_error = self.mols * np.sqrt((self.grams_error / self.grams)**2)
330    
331    def set_cross_section(self):
332        '''Set the cross section of the material, based on the atoms dict.'''
333        total_cross_section = 0.0
334        for key in self.atoms:
335            total_cross_section += self.atoms[key] * cross_section[key]
336        self.cross_section = total_cross_section
